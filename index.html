Q1:Find the number of digits

Ans:

    const Find_Digits = (N) => 
{
	let ans=0;
	while(N!==0){
	  ans++;
	  N=Math.floor(N/10);
	}
	return ans;
};

{
let count =0;
while(N!==0){
  let check = N%10;
  if(check ===5){
    count++;
  }
  N=Math.floor(N/10);
}
return count;
};

    const findSum = (n) => {
        let ans=0;
        for(i=1;i<=n;i++)

Q4:Find the sum of the digits of a given number.

Ans:

    const Number_Sum = (N) => 
{
  const numStr = N.toString();
	let sum=0;
	for(const ch of numStr){
	  sum += Number(ch);
	}
	return sum;
};

Q6:Print the Pattern.

Ans:

    const Print_pattern = (N) => 
{
let star='';
  for(let i=1;i<=N;i++){
    for(let j=1;j<=i;j++){
      star=star+'*';
    }
    star=star+"\n";
  }
console.log(star);
};
